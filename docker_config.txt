server {
    listen 80;
    server_name homedesigner.space www.homedesigner.space;

    # Redirect HTTP to HTTPS
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name www.homedesigner.space homedesigner.space;

    root /var/www/html/projects/static/homedesigner.space;
    index index.html index.htm;

    ssl_certificate /etc/letsencrypt/live/www.homedesigner.space/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/www.homedesigner.space/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

    access_log /var/log/nginx/homedesigner.space.ssl.access.log;
    error_log /var/log/nginx/homedesigner.space.ssl.error.log;

    # Proxy all traffic to your Docker container's application on port 8111
    location / {
        proxy_pass http://localhost:8111;  # Update to match the port where your app is running
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
